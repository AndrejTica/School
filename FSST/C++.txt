#include <iostream>
using namespace std;

class Dummy {
  public:
    int isitme (Dummy& param);     //why not (CDummy* param);
};

int Dummy::isitme (Dummy& param)  //we are giving the function an object as parameter so that the function can use its contents 
{
  if (&param == this) return true;
  else return false;
}

class Dummy2 {
public:
    int isitme2 (Dummy2* param);
};


int Dummy2::isitme2 (Dummy2* param)
{
    if (param == this) return true;
    else return false;
}



int main () {
  Dummy a;
  Dummy* b = &a;
  if ( b->isitme(a) )               //why not isitme(&a)
    cout << "yes, &a is b" << endl;



  Dummy2 x;
  Dummy2* y = &x;
  if ( y->isitme2(&x) )               //why not isitme(&a)
    cout << "yes, &x is y";
  return 0;


}
