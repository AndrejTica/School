8. Was gibt der folg. Algorithmus aus (für m=2 und n= 3)
	{
		m=2;
		n= 3;
		i= 1;
		j= m;
		while (i != n) {
			i= i + 1;
			j= j * m;
		}
		j ausgeben
	}
