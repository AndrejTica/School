Hinweis: strtok
	Mit der Funktion strtok() können Sie einen String in einzelne 
	Teilstrings anhand von Tokens zerlegen. Zuerst die Syntax:
	
	char *strtok(char *s1, const char *s2);
	
	Damit wird der String s1 durch das Token getrennt, welches 
	sich im s2 befindet. Ein Token ist ein String, der keine 
	Zeichen aus s2 enthält. Ein Beispiel:
	
	/* strtok.c */
	#include <stdio.h>
	#include <stdlib.h>
	#include <string.h>
	int main(void) {
	   char string[] = "Ein Teststring mit mehreren Worten\n"
					   "und mehreren Zeilen.\t Ende\n";
	   int i=1;
	   char *ptr;
	   ptr = strtok(string, "\n\t ");
	   while(ptr != NULL) {
		  printf("% d. Wort: %s\n",i++,ptr);
		  ptr = strtok(NULL, "\n\t ");
	   }
	   return EXIT_SUCCESS;
	}
