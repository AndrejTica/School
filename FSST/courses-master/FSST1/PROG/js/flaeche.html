Aufgabe: Flaeche

Lies laenge und breite. Berechne die Fläche des Rechtecks.

function calc(){
	console.log("calc called");
	
	var slaenge= document.getElementById("laenge").value;
	var sbreite= document.getElementById("breite").value;
	var flaenge= parseFloat(slaenge);
	var fbreite= parseFloat(sbreite);
	
	var flaeche= flaenge * fbreite;
	
	var sout= "Flaeche= " + flaeche;
	
	
	document.getElementById("ausgabe").innerHTML = "" + sout + "";
	
	console.log(sout);
}
