public void actionPerformed(ActionEvent arg0) {
				
				final String username = "valentina.tollinger@gmail.com";		//Eigene Email Adresse eingeben
				final String password = "!!!!!!!!!!!!!!!"; 				//Passwort einfügen
				final String recipient = "anton.hofmann@htl-salzburg.ac.at";		//Empfänger Email Adresse eingeben



				Properties props = new Properties();					
				props.put("mail.smtp.starttls.enable", "true");
				props.put("mail.smtp.auth", "true");
				props.put("mail.smtp.host", "smtp.gmail.com");				//SMTP-Server aktualisieren
				props.put("mail.smtp.port", "587");					//richtige Portnummer eingeben

				Session session = Session.getInstance(props,
						new javax.mail.Authenticator() {
					protected PasswordAuthentication getPasswordAuthentication() {
						return new PasswordAuthentication(username, password);
					}
				});

				try {

					Message message = new MimeMessage(session);
					message.setFrom(new InternetAddress(username));
					message.setRecipients(Message.RecipientType.TO,
							InternetAddress.parse(recipient));
					message.setSubject("!!!!!");					//Betreff eingeben
					message.setText("!!!!!");					//Text eingeben

					Transport.send(message);

					System.out.println("Done");

				} catch (MessagingException e) {
					throw new RuntimeException(e);
				}

				


			}
