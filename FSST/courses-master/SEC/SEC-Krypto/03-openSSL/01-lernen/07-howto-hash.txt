## openssl: file.txt unterschreiben und anschliessend verifizieren,
##			d.h. a) ein signature.sign mit dem privatekey erzeugen
##          b) signature.sign mit dem publickey verfizieren
-------------------------------------------------------------------

### To sign a file using SHA-256 with binary file output:
openssl dgst -sha256 -sign privatekey.pem -out signature.sign file.txt

### To verify a signature:
openssl dgst -sha256 -verify publickey.pem \
      -signature signature.sign file.txt



### To create a hex-encoded message digest of a file:
openssl dgst -sha256 -hex file.txt
